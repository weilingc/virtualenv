# virtualenv

pip install virtualenv

# 方法1 ----- 下語法創建資料夾
$ virtualenv xxx_env

# 方法2 ----- 創建資料夾後 加入venv資料夾
$ python3 -m venv tutorial-env


# Mac:
$ source xxx_env/bin/activate

# window:
自己測試可運行指令:
$ .venv\Scripts\activate
